1function [thrust_magnitude,desired_attitude] = fcn(R,ex,ev,a,psi,kx,kv,quadModel)
 2
 3e3 = [0 0 1]';
 4
 5desired_thrust = -kx*ex - kv*ev - quadModel.mass*quadModel.g*e3 + quadModel.mass*a;
 6
 7% Desired Attitude Calculation
 8z_direction = -desired_thrust/norm(desired_thrust); % b3, from [0 0 1]'
 9x_direction = findDirection(psi, z_direction, 0); % b1, from [1 0 0]'
10y_direction = hat(z_direction)*x_direction; % b2, from [0 1 0]'
11desired_attitude = [x_direction; y_direction; z_direction];
12thrust_magnitude = norm(desired_thrust) * (e3'*R'*desired_attitude*e3);
13
14end

16% Function to find Promoted Direction Vector
17function y = findDirection(psi,thrust_direction,initial_pitch)
18Rpsi = angle2dcm(0,initial_pitch,psi,123);
19ddv = Rpsi*[1 0 0]'; % Provisional Desired Direction Vector (DDV)
20pdv = - (hat(thrust_direction) * (hat(thrust_direction) * ddv)); % Promoted Direction Vector (PDV)
21if (isalmost(pdv,zeros(3,1),1e-3))
22    Rpsi = angle2dcm(0,initial_pitch,psi,123);
23    ddv = Rpsi*[1 0 0]'; % Provisional Desired Direction Vector (DDV)
24    pdv = - (hat(thrust_direction) * (hat(thrust_direction) * ddv)); % Promoted Direction Vector (PDV)
25    y = pdv/norm(pdv);
26else
27    y = pdv/norm(pdv);
28end
29end
